toCMYK : function (){
  var var_K = 1

  if ( this.c < var_K )   var_K = this.c;
  if ( this.m < var_K )   var_K = this.m;
  if ( this.y < var_K )   var_K = this.y;
  if ( var_K == 1 ) {
     var C = 0;
     var M = 0;
     var Y = 0;
  }
  else {
     var C = ( this.c - var_K ) / ( 1 - var_K );
     var M = ( this.m - var_K ) / ( 1 - var_K );
     var Y = ( this.y - var_K ) / ( 1 - var_K );
  }
  var K = var_K;
  return new CMYK(C,M,Y,K);
}
