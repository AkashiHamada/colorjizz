public function websafe() {
    $c = array('00', 'CC', '33', '66', '99', 'FF');
    $palette = array();
    for ($i = 0; $i < 6; $i++) {
        for ($j = 0; $j < 6; $j++) {
            for ($k = 0; $k < 6; $k++) {
                $palette[] = new Hex($c[$i] + $c[$j] + $c[$k]);
            }
        }
    }
    return $this->match($palette);
}
