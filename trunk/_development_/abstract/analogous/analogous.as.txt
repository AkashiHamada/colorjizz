public function analogous(includeSelf:Boolean = false):Vector.<AbstractColor>
{                       
  var rtn:Vector.<AbstractColor> = new Vector.<AbstractColor>();
  rtn.push(this.hue(30)[this.toSelf]());
  if (includeSelf){
   rtn.push(this);
  }
  rtn.push(this.hue(-30)[this.toSelf]());
  return rtn;
}
