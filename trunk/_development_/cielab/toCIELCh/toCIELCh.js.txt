toCIELCh : function (){
		var var_H = Math.atan2( this.b, this.a );

		if ( var_H > 0 ) {
			var_H = ( var_H / Math.PI ) * 180;
		}else{
			var_H = 360 - ( Math.abs( var_H ) / Math.PI ) * 180
		}

		var l = this.l;
		var c = Math.sqrt(Math.pow(this.a,2) + Math.pow(this.b,2));
		var h = var_H;

		return new CIELCh(l,c,h);
	}
