public function toCIELCh() {
    $var_H = atan2($this->b, $this->a);

    if ($var_H > 0) {
        $var_H = ($var_H / pi()) * 180;
    } else {
        $var_H = 360 - (abs($var_H) / pi()) * 180;
    }

    $l = $this->l;
    $c = sqrt(pow($this->a, 2) + pow($this->b, 2));
    $h = $var_H;

    return new CIELCh($l, $c, $h);
}
