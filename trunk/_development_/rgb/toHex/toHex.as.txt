override public function toHex():Hex {
  var rgbcols:Array = new Array(this.r,this.g,this.b);
  var hex:String = '';
  for (var i:int=0;i<rgbcols.length;i++){
    var chara:String = "0123456789ABCDEF";
    hex += chara.charAt(Math.floor(rgbcols[i] / 16)) + chara.charAt(rgbcols[i] - (Math.floor(rgbcols[i] / 16) * 16));
  }
  return Hex.fromString(hex);
}
